32  def create
  
33    #Si tiene un archivo e input entonces
  
34    if params[:project][:input_url]
  
35      # Creamos todo el arbol de tareas sin crear las tareas en el kanban,
  
36      # ya que requeriria crear muchos objetos del kanban que luego se iran borrando
  
37      Rails.application.config.allow_task_position = false
  
38      i = Importer.new(params[:project][:input_url].path())
  
39      proyecto = i.import
  
40      current_user.projects << proyecto
  
41
  
42      #Ahora decimos que si se pueda llamar a task position y por lo tanto crear los objetos del kanban
  
43      Rails.application.config.allow_task_position = true
  
44
  
45      proyecto.create_kanban
  
46
  
47      redirect_to projects_path
  
48
  
49    else
  
50      @project = Project.new(project_params)
  
51      respond_to do |format|
  
52        if @project.save
  
53          #buscamos al (o los) super_admin
  
54          @user = User.where(:role => "super_admin")
  
55
  
56          #le(s) asignamos el proyecto recien creado
  
57          @user.each  do |user|
  
58          user.projects << @project
  
59          end
  
60          format.html { redirect_to @project}
  
61          format.json { render :show, status: :created, location: @project }
  
62        else
  
63          format.html { render :new}
  
64          format.json { render json: @project.errors, status: :unprocessable_entity }
  
65        end
  
66      end
  
67    end
